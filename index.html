// ================ ملف واحد كامل: app.js ================ // يتطلب تثبيت: npm install express mongoose cors jsonwebtoken dotenv require("dotenv").config(); const express = require("express"); const cors = require("cors"); const mongoose = require("mongoose"); const jwt = require("jsonwebtoken"); // ======================== إعدادات السيرفر ======================== const app = express(); app.use(cors()); app.use(express.json()); // ======================== الاتصال بقاعدة البيانات ======================== const connectDB = async () => { try { await mongoose.connect(process.env.MONGO_URI || "mongodb://localhost:27017/drmedia"); console.log("MongoDB Connected"); } catch (err) { console.error("DB connection error:", err); process.exit(1); } }; connectDB(); // ======================== النماذج (Models) ======================== // نموذج الشركة const companySchema = new mongoose.Schema({ name: { type: String, required: true }, plan: { type: String, default: "free" }, createdAt: { type: Date, default: Date.now } }); const Company = mongoose.model("Company", companySchema); // نموذج المستخدم const userSchema = new mongoose.Schema({ username: { type: String, required: true, unique: true }, password: { type: String, required: true }, role: { type: String, default: "employee" }, companyId: { type: mongoose.Schema.Types.ObjectId, ref: "Company", required: true }, employeeId: { type: mongoose.Schema.Types.ObjectId, ref: "Employee" } }); const User = mongoose.model("User", userSchema); // نموذج الحجز const bookingSchema = new mongoose.Schema({ title: String, date: Date, hallId: { type: mongoose.Schema.Types.ObjectId, ref: "Hall" }, employeeId: { type: mongoose.Schema.Types.ObjectId, ref: "Employee" }, companyId: { type: mongoose.Schema.Types.ObjectId, ref: "Company", required: true } }); const Booking = mongoose.model("Booking", bookingSchema); // نموذج الموظف const employeeSchema = new mongoose.Schema({ name: String, email: String, phone: String, companyId: { type: mongoose.Schema.Types.ObjectId, ref: "Company", required: true } }); const Employee = mongoose.model("Employee", employeeSchema); // نموذج القاعة const hallSchema = new mongoose.Schema({ name: String, capacity: Number, price: Number, companyId: { type: mongoose.Schema.Types.ObjectId, ref: "Company", required: true } }); const Hall = mongoose.model("Hall", hallSchema); // ======================== ميدلوير المصادقة (JWT) ======================== const auth = (req, res, next) => { const token = req.header("Authorization"); if (!token) return res.status(401).json({ msg: "No token, authorization denied" }); try { const decoded = jwt.verify(token.replace("Bearer ", ""), process.env.JWT_SECRET || "SECRET_KEY"); req.user = decoded; next(); } catch (e) { res.status(401).json({ msg: "Invalid token" }); } }; // ======================== المسارات (Routes) ======================== // 1) مسارات المصادقة app.post("/api/auth/login", async (req, res) => { const { username, password } = req.body; const user = await User.findOne({ username, password }); if (!user) return res.status(400).json({ msg: "Invalid credentials" }); const token = jwt.sign( { id: user._id, role: user.role, companyId: user.companyId }, process.env.JWT_SECRET || "SECRET_KEY" ); res.json({ token, user: { id: user._id, username, role: user.role, companyId: user.companyId } }); }); app.post("/api/auth/register", async (req, res) => { const { username, password, role, companyId, employeeId } = req.body; const existing = await User.findOne({ username }); if (existing) return res.status(400).json({ msg: "Username exists" }); const user = new User({ username, password, role, companyId, employeeId }); await user.save(); res.json({ msg: "User created", user }); }); // 2) مسارات الحجوزات (محمية) app.get("/api/bookings", auth, async (req, res) => { const bookings = await Booking.find({ companyId: req.user.companyId }).populate("hallId employeeId"); res.json(bookings); }); app.post("/api/bookings", auth, async (req, res) => { const booking = new Booking({ ...req.body, companyId: req.user.companyId }); await booking.save(); res.json(booking); }); app.delete("/api/bookings/:id", auth, async (req, res) => { await Booking.findOneAndDelete({ _id: req.params.id, companyId: req.user.companyId }); res.json({ ok: true }); }); // 3) مسارات الموظفين (محمية) app.get("/api/employees", auth, async (req, res) => { const employees = await Employee.find({ companyId: req.user.companyId }); res.json(employees); }); app.post("/api/employees", auth, async (req, res) => { const employee = new Employee({ ...req.body, companyId: req.user.companyId }); await employee.save(); res.json(employee); }); // 4) مسارات القاعات (محمية) app.get("/api/halls", auth, async (req, res) => { const halls = await Hall.find({ companyId: req.user.companyId }); res.json(halls); }); app.post("/api/halls", auth, async (req, res) => { const hall = new Hall({ ...req.body, companyId: req.user.companyId }); await hall.save(); res.json(hall); }); // مسار اختبار app.get("/", (req, res) => res.send("Drmedia SaaS API Running")); // ======================== تشغيل السيرفر ======================== const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
